```js
    function startPhase2() {
      // ── Show Rio canvas, hide Phase 1 ──
      window.overlayRestart = startPhase2;
      renderer.domElement.style.display = 'none';
      document.getElementById('hud').style.display = 'none';
      document.getElementById('overlay').classList.remove('show');
      document.getElementById('overlay-play-again-btn').style.display = 'none';
      document.getElementById('choose-level-btn').style.display   = 'none';

      const wrap = document.getElementById('p2-canvas-wrap');

      wrap.classList.add('show');

      // ── Canvas setup ──
      const p2canvas = document.createElement('canvas');
      wrap.appendChild(p2canvas);
      const p2ctx = p2canvas.getContext('2d');
      const CW = 510, CH = 444;
      let p2scale = 1, p2ox = 0, p2oy = 0;

      function p2Resize() {
        p2canvas.width  = window.innerWidth;
        p2canvas.height = window.innerHeight;
        p2scale = Math.min(window.innerWidth / CW, window.innerHeight / CH);
        p2ox = (window.innerWidth  - CW * p2scale) / 2;
        p2oy = (window.innerHeight - CH * p2scale) / 2;
      }
      p2Resize();
      window.addEventListener('resize', p2Resize);

      // Convert browser clientX/Y → canvas logical coords
      function toLogical(cx, cy) {
        return { x: (cx - p2ox) / p2scale, y: (cy - p2oy) / p2scale };

      const ROOMS = [

      // Shared walls: rooms touching directly with a doorway opening
      const SHARED_WALLS = [
        {y:206, x:30,  w:22, a:1,  b:2 },
        {y:380, x:30,  w:22, a:3,  b:4 },
        {y:68,  x:162, w:22, a:5,  b:6 },

      // ── Passability ──
      function isPassable(px, py) {
        for (const r of ROOMS)
          if (px >= r.x && px <= r.x+r.w && py >= r.y && py <= r.y+r.h) return true;
        for (const c of CORRIDORS)
          if (px >= c.x && px <= c.x+c.w && py >= c.y && py <= c.y+c.h) return true;
        return false;
      }

      function roomContaining(px, py) {
        for (const r of ROOMS)
          if (px >= r.x && px <= r.x+r.w && py >= r.y && py <= r.y+r.h) return r;
        return null;
      }

      // ── Wall drawing helpers ──
      function openingsFor(r) {
        const op = {top:[],bottom:[],left:[],right:[]};
        for (const c of CORRIDORS) {
          const H = c.w > c.h;
          if (H) {
            if (c.x     === r.x+r.w && c.y>=r.y && c.y+c.h<=r.y+r.h) op.right.push({from:c.y,to:c.y+c.h});
            if (c.x+c.w === r.x     && c.y>=r.y && c.y+c.h<=r.y+r.h) op.left.push ({from:c.y,to:c.y+c.h});
          } else {
            if (c.y     === r.y+r.h && c.x>=r.x && c.x+c.w<=r.x+r.w) op.bottom.push({from:c.x,to:c.x+c.w});
            if (c.y+c.h === r.y     && c.x>=r.x && c.x+c.w<=r.x+r.w) op.top.push   ({from:c.x,to:c.x+c.w});
          }
        }
        for (const sw of SHARED_WALLS) {
          if (sw.a === r.id) op.bottom.push({from:sw.x, to:sw.x+sw.w});
          if (sw.b === r.id) op.top.push   ({from:sw.x, to:sw.x+sw.w});
        }
        return op;
      }

      function segs(ctx, horiz, fixed, start, end, openings) {
        const sorted = [...openings].sort((a,b)=>a.from-b.from);
        let pos = start;
        ctx.beginPath();
        for (const {from,to} of sorted) {
          if (pos < from) horiz ? (ctx.moveTo(pos,fixed),ctx.lineTo(from,fixed)) : (ctx.moveTo(fixed,pos),ctx.lineTo(fixed,from));
          pos = Math.max(pos, to);
        }
        if (pos < end) horiz ? (ctx.moveTo(pos,fixed),ctx.lineTo(end,fixed)) : (ctx.moveTo(fixed,pos),ctx.lineTo(fixed,end));
        ctx.stroke();
      }

      function drawRoomWalls(ctx, r) {
        const {x,y,w,h} = r, op = openingsFor(r);
        segs(ctx,true, y,   x,   x+w, op.top);
        segs(ctx,true, y+h, x,   x+w, op.bottom);
        segs(ctx,false,x,   y,   y+h, op.left);
        segs(ctx,false,x+w, y,   y+h, op.right);
      }

      function drawCorrWalls(ctx, c) {
        const H = c.w > c.h;
        ctx.beginPath();
        if (H) {
          ctx.moveTo(c.x,c.y);     ctx.lineTo(c.x+c.w,c.y);
          ctx.moveTo(c.x,c.y+c.h); ctx.lineTo(c.x+c.w,c.y+c.h);
        } else {
          ctx.moveTo(c.x,c.y);     ctx.lineTo(c.x,c.y+c.h);
          ctx.moveTo(c.x+c.w,c.y); ctx.lineTo(c.x+c.w,c.y+c.h);
        }
        ctx.stroke();
      }

      // ── Draw map (static for now — fog added in Task 5) ──
      function drawMap(ctx, visitedRooms) {
        ctx.fillStyle = '#07070f';
        ctx.fillRect(0, 0, CW, CH);

        // Floors
        for (const r of ROOMS) {
          if (visitedRooms && !visitedRooms.has(r.id)) continue;
          ctx.fillStyle = r.type==='exit'?'#071a09':r.type==='dead'?'#180a0a':'#10102a';
          ctx.fillRect(r.x, r.y, r.w, r.h);
        }
        for (const c of CORRIDORS) {
          if (visitedRooms && !visitedRooms.has(c.rooms[0]) && !visitedRooms.has(c.rooms[1])) continue;
          ctx.fillStyle = '#0d0d20';
          ctx.fillRect(c.x, c.y, c.w, c.h);
        }

        // Walls
        for (const r of ROOMS) {
          if (visitedRooms && !visitedRooms.has(r.id)) continue;
          ctx.lineWidth = 1.5;
          ctx.strokeStyle = r.type==='exit'?'#006633':r.type==='dead'?'#3a1010':'#222248';
          drawRoomWalls(ctx, r);
        }
        ctx.strokeStyle = '#181830';
        ctx.lineWidth = 1;
        for (const c of CORRIDORS) {
          if (visitedRooms && !visitedRooms.has(c.rooms[0]) && !visitedRooms.has(c.rooms[1])) continue;
          drawCorrWalls(ctx, c);
        }

        // Labels (START / ✕ / ✦)
        ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
        for (const r of ROOMS) {
          if (visitedRooms && !visitedRooms.has(r.id)) continue;
          const cx = r.x+r.w/2, cy = r.y+r.h/2;
          if (r.type==='exit')   { ctx.fillStyle='#00cc55'; ctx.font='bold 11px monospace'; ctx.fillText('✦ EXIT',cx,cy); }
          else if (r.type==='dead')   { ctx.fillStyle='#773333'; ctx.font='bold 10px monospace'; ctx.fillText('✕',cx,cy); }
          else if (r.type==='start')  { ctx.fillStyle='#3355aa'; ctx.font='bold 9px monospace';  ctx.fillText('START',cx,cy); }
        }
      }

      // ── Static test render ──
      p2ctx.setTransform(p2scale, 0, 0, p2scale, p2ox, p2oy);
      drawMap(p2ctx, null); // null = show all rooms (no fog)
    }
```

```js
      const R0 = ROOMS[0];
      const player = {
        x: R0.x + R0.w / 2,

        angle: 0,       // radians, 0 = right

      window.addEventListener('keydown', e => { keys[e.code] = true;  });
      window.addEventListener('keyup',   e => { keys[e.code] = false; });

      function p2Loop() {

        if (p2Over) {

          drawMap(p2ctx, visitedRooms);
          drawPlayer(p2ctx);
          return;
        }

        const now = performance.now();
        const dt  = Math.min((now - p2Last) / 1000, 0.05);
        p2Last = now;

        // ── Player movement ──
        player.angle = Math.atan2(
          mouseLogical.y - player.y,
          mouseLogical.x - player.x
        );

        let dx = 0, dy = 0;
        if (keys['KeyW'] || keys['ArrowUp'])    dy -= 1;
        if (keys['KeyS'] || keys['ArrowDown'])  dy += 1;
        if (keys['KeyA'] || keys['ArrowLeft'])  dx -= 1;
        if (keys['KeyD'] || keys['ArrowRight']) dx += 1;
        if (dx !== 0 && dy !== 0) { dx *= 0.707; dy *= 0.707; }
        player.x += dx * player.speed * dt;
        player.y += dy * player.speed * dt;

        // ── Render ──
        p2ctx.setTransform(p2scale, 0, 0, p2scale, p2ox, p2oy);
        drawMap(p2ctx, visitedRooms);
        drawPlayer(p2ctx);
      }
      p2Loop();
    }
```

```js
      // Line-of-sight: sample 5 points from enemy toward player
      function hasLOS(ex, ey, px, py) {
        for (let t = 0.1; t <= 0.9; t += 0.2) {
          if (!isPassable(ex + (px-ex)*t, ey + (py-ey)*t)) return false;
        }
        return true;
      }

      // Returns true if player is inside enemy's 30° FOV cone at range ≤150
      function canSeePlayer(e) {
        const dx = player.x - e.x, dy = player.y - e.y;
        const dist = Math.sqrt(dx*dx + dy*dy);
        if (dist > 150) return false;
        const angleToPlayer = Math.atan2(dy, dx);
        let diff = angleToPlayer - e.angle;
        // Normalise to [-π, π]
        while (diff >  Math.PI) diff -= Math.PI*2;
        while (diff < -Math.PI) diff += Math.PI*2;
        if (Math.abs(diff) > Math.PI*15/180) return false; // 30° cone = ±15°
        return hasLOS(e.x, e.y, player.x, player.y);
      }
```

```js
        for (const e of enemies) {

              e.angle = e.patrolDir > 0 ? 0 : Math.PI;
```
